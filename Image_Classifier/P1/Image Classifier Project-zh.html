class Network(nn.Module):
    def __init__(self, input_size, output_size, hidden, drop_p=0.5):
        super().__init__()
        self.hidden_layers = nn.ModuleList([nn.Linear(input_size, hidden[0])])
        
        layer_sizes = zip(hidden[:-1], hidden[1:])
        self.hidden_layers.extend([nn.Linear(h1, h2) for h1, h2 in layer_sizes])
        
        self.output = nn.Linear(hidden[-1], output_size)
        
        self.dropout = nn.Dropout(p=drop_p)
        
    def forward(self, x):
        
        for each in self.hidden_layers:
            x = F.relu(each(x))
            x = self.dropout(x)
        x = self.output(x)
        
        return F.log_softmax(x, dim=1)

def Validation(model,data_loader,criterion,device='cpu'):
    loss = 0
    accuracy = 0
    
    for data in data_loader:
        images,labels = data
        
        images,labels = images.to(device),labels.to(device)
        
        output = model.forward(images)
        loss += criterion(output,labels)
        
        ps = torch.exp(output)
        equality = (labels.data == ps.max(dim=1)[1])
        accuracy += equality.type(torch.FloatTensor).mean()
        
    return loss,accuracy

def train_leanning(model,criterion,optimizer,data_loader,epochs=3,print_every=30,device='cpu'):
    train_loss = 0
    steps = 0
    train_loss = 0
    
    for e in range(epochs):
        # 
        model.train()
        
        train_loss = 0
        model.to(device)
        
        for ii,(images,labels) in enumerate(data_loader):
            steps += 1
            
            images,labels = images.to(device),labels.to(device)
            
            optimizer.zero_grad()
            
            output = model.forward(images)
            loss = criterion(output,labels)
            loss.backward()
            optimizer.step()
            
            train_loss += loss.item()
            
            if steps % print_every == 0:
                # 
                model.eval()
                
                with torch.no_grad():
                    val_loss,accuracy = Validation(model,validation_dataloader,criterion,device)
                    
                print("device : {},".format(device),
                      "epochs : {}/{},".format(e+1,epochs),
                      "validation loss : {:.3f},".format(val_loss/len(validation_dataloader)),
                      "validation accuracy : {:.3f},".format(accuracy/len(validation_dataloader)),
                      "train loss : {:.4f}".format(train_loss/print_every))
                
                train_loss = 0
                
                model.train()

def load_checkpoint(filepath):
    check_point = torch.load(filepath)
    
    # 
    model = models.vgg19()
    
    for params in model.parameters():
        params.requires_grad = False
    
    # classifier classifier
    classifier = Network(input_size,output_size,hidden)
    model.classifier = classifier
    
    # model  state_dict 
    model.load_state_dict(check_point['state_dict'])
    
    # optimizer  state_dict
    optimizer = optim.Adam(model.classifier.parameters(),lr=0.0001)
    optimizer.load_state_dict(check_point['optimizer'])
    
    # train  class_to_idx
    image_class_idx = check_point['image_class_idx']
    
    return model,optimizer,image_class_idx

import numpy as np

from PIL import Image
from matplotlib import pyplot as plt

def process_image(image):
    ''' Scales, crops, and normalizes a PIL image for a PyTorch model,
        returns an Numpy array
    '''
    
    # TODO: Process a PIL image for use in a PyTorch model
    mean = np.array([0.485, 0.456, 0.406])
    std = np.array([0.229, 0.224, 0.225]) 
    
    #  224x224 
    img = Image.open(image)
    img_256 = img.resize(size=(256,256))
    img_224 = img_256.crop(box=(0,0,224,224))
    
    #
    np_image = np.array(img_224)
    np_image = np_image.astype(float)
    
    #
    np_image = (np_image - mean)/std
    
    #
    np_image_T = np_image.transpose((2,0,1))
    
    #plt.subplot(1,2,1)
    #plt.title("img 256 * 256")
    #plt.imshow(img_256)
   
    #plt.subplot(1,2,2)
    #plt.title("img 224 * 224")
    #plt.imshow(img_224)
    
    return np_image_T

def imshow(image, ax=None, title=None):
    """Imshow for Tensor."""
    if ax is None:
        fig, ax = plt.subplots()
    
    # PyTorch tensors assume the color channel is the first dimension
    # but matplotlib assumes is the third dimension
    image = image.numpy().transpose((1, 2, 0))
    
    # Undo preprocessing
    mean = np.array([0.485, 0.456, 0.406])
    std = np.array([0.229, 0.224, 0.225])
    image = std * image + mean
    
    # Image needs to be clipped between 0 and 1 or it looks like noise when displayed
    image = np.clip(image, 0, 1)
    
    ax.imshow(image)
    
    return ax
